[docs]def black_and_white(lut):
    """
    creates a black and white lut
    """
    lut.alpha_range=[1.,1.]
    lut.hue_range=[0.,1.]
    lut.value_range=[1.0, 2.0]
    lut.saturation_range=[0.,0.]
    return lut
