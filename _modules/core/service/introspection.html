[docs]def name(obj):
    if hasattr(obj, 'name'):
        return obj.name
    elif hasattr(obj, '__class__'):
        return obj.__class__.__name__
    elif hasattr(obj, '__name__'):
        return obj.__name__
    else:
        return str(obj)




[docs]def alias(obj):
    if hasattr(obj, '__alias__'):
        return obj.__alias__
    elif hasattr(obj, 'alias'):
        return obj.alias
    else:
        return str(name(obj)).capitalize()
