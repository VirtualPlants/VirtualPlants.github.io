[docs]    def default(cls):
        return (0, 255)


    def __repr__(self):
        default_min = -2 ** 24
        default_max = 2 ** 24
        if (self.min == default_min and
                self.max == default_max):
            return self.__class__.__name__
        else:
            return 'IIntRange(min=%d, max=%d)' % (self.min, self.max)
