        if shape is None:
            shape = 'hline'

        if shape == 'hline':
            widget = IntSlider()
        elif shape == 'vline':
            widget = IntSimpleSlider()
            widget.setOrientation(QtCore.Qt.Vertical)
        elif shape in ('large', 'small', 'responsive'):
            widget = IntDial()
        else:
            widget = None
        return widget
