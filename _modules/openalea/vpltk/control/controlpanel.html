[docs]    def add(self, ctrl):
        row = self.rowCount()
        self.insertRow(row)
        
        manager = get_manager(type(ctrl))
        if manager is not None:
            self.setItem(row,0,manager.displayThumbnail())
        else:
            self.setItem(row,0,QtGui.QTableWidgetItem(str(ctrl)))
