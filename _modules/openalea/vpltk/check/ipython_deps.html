[docs]def has_full_deps():
    """
    Check if User can use IPython shell embeded in OpenAlea.
    
    Check zmq, pygments
    
    :return: True if user has zmq and pygments. Else False.
    """
    # Check Python ZeroMQ
    zmq = has_zmq()
    # Check Pygments
    pgm = has_pygments()
	
    return zmq & pgm
